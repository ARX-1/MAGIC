def load_file(filename):
    subor=open(mode="r", encoding="UTF8", file=f"{filename}.txt")
    my_list=[]
    for i in subor:
        my_list+=[i[:-1]]
    return(my_list)
    
def split_list(listname):
    split_list=[]
    for prvok in listname:
        prvok=prvok+" "
        position=0
        for char in range(len(prvok)):
            if prvok[char]==" ":
                split_list+=[prvok[position:char]]
                position=char+1
    return(split_list)
